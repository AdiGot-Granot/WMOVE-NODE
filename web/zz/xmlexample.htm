The Example: Here is the DocType 
<?xml version="1.0"?>
<!DOCTYPE xdoc [
<!ELEMENT xdoc (moving)>
<!ELEMENT moving (inv)*>
<!ELEMENT inv (recnum,item,letter,cuft,lbs,qty,code,volume)>
<!ELEMENT recnum (#PCDATA)>
<!ATTLIST recnum
          type CDATA #FIXED "float"
          size CDATA #FIXED "11"
>
<!ELEMENT item (#PCDATA)>
<!ATTLIST item
          type CDATA #FIXED "string"
          size CDATA #FIXED "30"
>
<!ELEMENT letter (#PCDATA)>
<!ATTLIST letter
          type CDATA #FIXED "string"
          size CDATA #FIXED "1"
>
<!ELEMENT cuft (#PCDATA)>
<!ATTLIST cuft
          type CDATA #FIXED "float"
          size CDATA #FIXED "7"
          precision CDATA #FIXED "2"
>
<!ELEMENT lbs (#PCDATA)>
<!ATTLIST lbs
          type CDATA #FIXED "float"
          size CDATA #FIXED "8"
          precision CDATA #FIXED "2"
>
<!ELEMENT qty (#PCDATA)>
<!ATTLIST qty
          type CDATA #FIXED "float"
          size CDATA #FIXED "3"
>
<!ELEMENT code (#PCDATA)>
<!ATTLIST code
          type CDATA #FIXED "string"
          size CDATA #FIXED "2"
>
<!ELEMENT volume (#PCDATA)>
<!ATTLIST volume
          type CDATA #FIXED "string"
          size CDATA #FIXED "15"
>

]>
